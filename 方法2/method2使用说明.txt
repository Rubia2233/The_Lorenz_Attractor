%simulink跑完之后复制下面代码即可
% 1. 清理残留变量
clear x1_m2 x2_m2 x3_m2;

% 2. 严格从 out 结构体中提取方法二的数据
x1_m2 = out.simout2(:, 1); 
x2_m2 = out.simout2(:, 2); 
x3_m2 = out.simout2(:, 3); 

% 3. 新开窗口绘制紫色的 3D 混沌蝴蝶
figure;
% [0.5, 0, 0.9] 是非常高饱和度、纯正的深紫色
plot3(x1_m2, x2_m2, x3_m2, 'Color', [0.5, 0, 0.9], 'LineWidth', 1.3); 
grid on;

% 4. 规范化坐标轴标签
xlabel('x_1'); ylabel('x_2'); zlabel('x_3');
title('Lorenz System 3D Phase Plane (Method 2 - Mystic Purple)');
